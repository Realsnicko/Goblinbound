from fastapi import FastAPI, HTTPException from fastapi.middleware.cors import CORSMiddleware from fastapi.responses import PlainTextResponse from api import get_inventory_data, format_inventory_text from SLH2 import load_player_sync from battle import preparebattle, run_battle app = FastAPI() # Normal CORS middleware (still needed) app.add_middleware( CORSMiddleware, allow_origins=["*"], allow_credentials=True, allow_methods=["*"], allow_headers=["*"], ) # FORCE CORS HEADERS ALWAYS @app.middleware("http") async def add_cors_headers(request, call_next): response = await call_next(request) response.headers["Access-Control-Allow-Origin"] = "*" response.headers["Access-Control-Allow-Headers"] = "*" response.headers["Access-Control-Allow-Methods"] = "*" return response @app.get("/") def rooot(): return {"status": "GoblinBound API Online"} @app.get("/inventory", response_class=PlainTextResponse) def inventory(uid: int): data = get_inventory_data(uid) return format_inventory_text(data) @app.get("/battle", response_class=PlainTextResponse) def battle(uid: int): try: data = load_player_sync(uid) player, enemy, _ = preparebattle(data, uid=uid) result = run_battle(player, enemy, uid=uid) return result["log"] except Exception as e: print("BATTLE ERROR:", e) raise HTTPException(status_code=500, detail=str(e))
